int qpow(int a,int p,int mod) {
    int t = 1;
    int y = a;
    println(toString(a) + " " + toString(p) + " " + toString(mod));
    while(p>0){
        if((p&1) == 1)t=t*y % mod;
        println(toString(p&1));
        y=y*y % mod;
        p=p / 2;
        println("p = " + toString(p) + " t = " + toString(t));
    }
    return t;
}
